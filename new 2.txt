CREATE OR REPLACE DIRECTORY my_csv_dir AS 'C:\oracle_csv';
GRANT READ, WRITE ON DIRECTORY my_csv_dir TO your_user;

CREATE TABLE ext_mydata (
  id        NUMBER,
  name      VARCHAR2(100),
  salary    NUMBER
)
ORGANIZATION EXTERNAL (
  TYPE ORACLE_LOADER
  DEFAULT DIRECTORY my_csv_dir
  ACCESS PARAMETERS (
    RECORDS DELIMITED BY NEWLINE
    FIELDS TERMINATED BY ',' OPTIONALLY ENCLOSED BY '"'
    MISSING FIELD VALUES ARE NULL
    (
      id, name, salary
    )
  )
  LOCATION ('mydata.csv')
)
REJECT LIMIT UNLIMITED;



SELECT * FROM ext_mydata;
