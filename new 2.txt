import aspose.pdf as ap

pdf_path = r"C:\path\to\input.pdf"
excel_path = r"C:\path\to\output.xlsx"

doc = ap.Document(pdf_path)

# --- Excel save options ---
options = ap.ExcelSaveOptions()
options.merge_adjacent_cells = True       # Merge visually continuous cells
options.minimize_the_number_of_worksheets = True  # ✅ key setting (keeps all pages in ONE sheet)
options.format = ap.ExcelSaveOptions.ExcelFormat.XLSX

# --- Save ---
doc.save(excel_path, options)
print("✅ PDF converted to single-sheet Excel file successfully!")
