SELECT 
    q1.as_of_date,
    q1.event_id,
    q1.event_description,
    q2.case_id
FROM
    (
        SELECT DISTINCT 
            as_of_date,
            event_id,
            event_description
        FROM eczrdkyc.raw_utr_cust_event_details
    ) q1
LEFT JOIN
    (
        SELECT DISTINCT 
            as_of_date,
            event_id,
            case_id
        FROM eczrdkyc.enriched_utr_case_status_details
    ) q2
ON q1.event_id = q2.event_id
LIMIT 20;
